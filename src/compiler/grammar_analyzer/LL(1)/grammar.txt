# 简单算术表达式文法
S' -> E
E -> T E'
E' -> + T E'
E' -> - T E'
E' -> ε
T -> F T'
T' -> * F T'
T' -> / F T'
T' -> ε
F -> id
F -> num
F -> ( E )

S -> while C do S

relop -> ==
relop -> !=
relop -> <
relop -> <=
relop -> >
